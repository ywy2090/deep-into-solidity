①
L1StandardBridge.bridgeETH()
// 用户入口，仅 EOA 可调用（onlyEOA 修饰符）
_minGasLimituint32// 用户指定的 L2 最小 gas 限制
_extraDatabytes// 附加数据（仅记录在事件中，不影响执行）
{ value: msg.value }// 附带的 ETH（即桥接金额）
→ 直接透传展开为 _initiateBridgeETH 的 5 个参数
_from= msg.sender // L1 发送方
_to= msg.sender // bridgeETH 时 from == to
_amount= msg.value // ETH 金额
_minGasLimit= _minGasLimit // 透传
_extraData= _extraData // 透传
调用 _initiateBridgeETH()
②
StandardBridge._initiateBridgeETH()
// 构造 finalizeBridgeETH 调用数据，通过 Messenger 发送
_fromaddress// L1 存款方地址
_toaddress// L2 接收方地址
_amountuint256// ETH 数量
_minGasLimituint32// 最小 gas 限制（透传）
_extraDatabytes// 附加数据（透传）
→ 将参数编码为 Level-1 消息体 _message（约 164 bytes）
_message= abi.encodeWithSelector(
  finalizeBridgeETH.selector, // 0x1635f5fd
  _from, _to, _amount, _extraData
)
调用 messenger.sendMessage{ value: _amount }(
  otherBridge, // L2StandardBridge 地址
  _message,
  _minGasLimit
)
调用 CrossDomainMessenger.sendMessage()
③
CrossDomainMessenger.sendMessage()
// 封装为 relayMessage 调用数据 + 计算 gas
_targetaddress// otherBridge（L2StandardBridge）
_messagebytes// finalizeBridgeETH 编码（来自 ②）
_minGasLimituint32// 用户指定的最小 gas（透传）
{ value: msg.value }// ETH 透传
→ 两个关键转换：① 生成版本化 nonce，② 计算 gasLimit，③ 编码 relayMessage
nonce= encodeVersionedNonce(msgNonce, 1)
// 高 16 位 = version(1)，低 240 位 = 递增序号
gasLimit= baseGas(_message, _minGasLimit)
// = TX_BASE_GAS
// + RELAY_CONSTANT_OVERHEAD (200_000)
// + RELAY_CALL_OVERHEAD (40_000)
// + RELAY_RESERVED_GAS (40_000)
// + _minGasLimit × 64/63 (EIP-150 缓冲)
// + calldata 字节费用
_data= abi.encodeWithSelector(
  relayMessage.selector, // 0xd764ad0b
  nonce,
  msg.sender, // = L1StandardBridge
  _target, // = L2StandardBridge
  msg.value, // ETH 金额
  _minGasLimit, // 用户指定 gas
  _message // finalizeBridgeETH 编码
)
调用 _sendMessage(
  otherMessenger, // L2CrossDomainMessenger
  gasLimit,
  msg.value,
  _data
)
调用 L1CrossDomainMessenger._sendMessage()
④
L1CrossDomainMessenger._sendMessage()
// 参数原封不动转发给 OptimismPortal.depositTransaction()
_toaddress// L2CrossDomainMessenger（otherMessenger）
_gasLimituint64// baseGas() 计算结果
_valueuint256// ETH（msg.value）
_databytes// relayMessage ABI 编码（来自 ③）
→ 参数一一映射到 portal.depositTransaction() 的签名
调用 portal.depositTransaction{ value: _value }(
  _to: L2CrossDomainMessenger,
  _value: _value, // 接收者得到的 ETH
  _gasLimit: _gasLimit, // L2 gas 上限
  _isCreation: false,
  _data: _data // relayMessage 编码
)
调用 OptimismPortal2.depositTransaction()
⑤
OptimismPortal2.depositTransaction()
// 打包 opaqueData，发出 TransactionDeposited 事件
_toaddress// L2CrossDomainMessenger
_valueuint256// 发送给接收者的 ETH
_gasLimituint64// L2 gas 上限
_isCreationbool// false（非合约创建）
_databytes// relayMessage ABI 编码
→ 打包 opaqueData（abi.encodePacked，紧凑非对齐），发出事件
opaqueData= abi.encodePacked(
  msg.value, // uint256，32 bytes，锁定总 ETH
  _value, // uint256，32 bytes，接收者 ETH
  _gasLimit, // uint64，8 bytes，L2 gas
  _isCreation,// bool，1 byte，0x00
  _data // bytes，变长，relayMessage 编码
)
emit TransactionDeposited(
  from, // alias(L1CrossDomainMessenger)
  _to, // L2CrossDomainMessenger
  version, // 0（DEPOSIT_VERSION）
  opaqueData // 上面打包的结果
)

▌ op-node 派生管道：解析 TransactionDeposited → 构造 L2 DepositTx
① 解析 opaqueData
msgValue:= opaqueData[0:32] // uint256
value:= opaqueData[32:64] // uint256
gasLimit:= opaqueData[64:72] // uint64
isCreate:= opaqueData[72] // bool
data:= opaqueData[73:] // bytes
② 构造 DepositTx（RLP 类型 0x7E）
SourceHash: hash(L1 区块 hash, logIndex)
From: from（事件 indexed 字段）
To: L2CrossDomainMessenger
Mint: msgValue（L2 铸造 ETH）
Value: value（发给 to）
Gas: gasLimit
Data: data（relayMessage 编码）
L2 执行：L2CrossDomainMessenger.relayMessage(nonce, sender, target, value, minGas, message)
→ 最终调用：L2StandardBridge.finalizeBridgeETH(_from, _to, _amount, _extraData)
